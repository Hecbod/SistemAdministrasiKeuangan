32         @Override
33         public void actionPerformed(ActionEvent ae) {
34             Object source = ae.getSource();
35             if (source == view.getBtnAddKonfirmasiPengajuan()){
36                 if (view.getTfIDPengajuanWR().getText().equals("")){
37                     JOptionPane.showMessageDialog(null, "Fill the ID");
38                 } else {
39                     PengajuanDana pd = model.cariIDPengajuan(view.getTfIDPengajuanWR().getText());
40                     if (pd != null){
41                         if (view.getRbSetujuPengajuan().isSelected()){
42                             if (pd.getTotal() > Integer.parseInt(view.getTfDanaUniversitas().getText())){
43                                 JOptionPane.showMessageDialog(null, "Sorry insufficient funds!");
44                             } else {
45                                 model.konfirmasiPengajuan(view.getTfIDPengajuanWR().getText(), "1");
46                             }
47                         } else {
48                             model.konfirmasiPengajuan(view.getTfIDPengajuanWR().getText(), "-1");
49                         }
50                         
51                     } else {
52                         JOptionPane.showMessageDialog(null, "Sorry ID can't be found!");
53                     }
54                 }
55             } else if (source == view.getBtnAddPembagianDana()){
56                 if (view.getTfFakultas().equals("") || view.getTfTotaLDana().equals("")){
57                     JOptionPane.showMessageDialog(null, "Fill all the textfield!");
58                 } else {
59                     if (model.cariFakultas(view.getTfFakultas().getText()) != null){
60                         PembagianDana pd = new PembagianDana(view.getTfPembagian().getText(), view.getTfFakultas().getText(), Integer.parseInt(view.getTfTotaLDana().getText()));
61                         Fakultas f = model.cariFakultas(view.getTfFakultas().getText());
62                         if (pd.getTotalDana() > Integer.parseInt(view.getTfDanaUniversitas().getText())){
63                             JOptionPane.showMessageDialog(null, "Sorry insufficient funds!");
64                         } else {
65                             model.tambahPembagianDana(pd);
66                             model.updateDanaFakultas(pd, f);
67                         }
68                     }
69                 }
70             } else if (source == view.getBtnLogout()){
71                 new ControllerLogin(model);
72                 view.dispose();
73             }
74             model.showTabelPengajuan(view);
75             view.getTfDanaUniversitas().setText(Integer.toString(model.showDana()));
76             view.refresh("");
77         }
