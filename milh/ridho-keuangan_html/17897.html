19         ControllerMenuFakultas(Aplikasi model, Fakultas f) {

34         @Override
35         public void actionPerformed(ActionEvent ae) {
36         Object source = ae.getSource();
37            
38             if (source == view.getBtnAddPengeluaran()){          
39                 if(!view.getTaKeterangan().getText().equals("")&&!view.getTfTotalPengeluaran().getText().equals("")){
40                     String idPengeluran = view.getTfIDPengeluaran().getText();
41                     String tahunAjar = (String)view.getComboTahunAjar().getSelectedItem();
42                     String taKeterangan = view.getTaKeterangan().getText();
43                     int totalPengeluaran = Integer.parseInt(view.getTfTotalPengeluaran().getText());
44                     if (model.showDanaSisa(fakultas.getIdFakultas())-totalPengeluaran>0){
45                         PengeluaranDana p = new PengeluaranDana(idPengeluran, fakultas.getIdFakultas(), tahunAjar, taKeterangan, totalPengeluaran);

49                         JOptionPane.showMessageDialog(null, "Not enough Dana that we have");
50                     } 
51                     view.getTaKeterangan().setText("");
52                     view.getTfTotalPengeluaran().setText("");
53                 }else{
54                     JOptionPane.showMessageDialog(null, "Fill all the data");
55                 }           
56             }else if (source == view.getBtnLihatKeterangan()){
57                 String keterangan = view.getTfLihatIDPengeluaran().getText();
58                 if(!view.getTfLihatIDPengeluaran().getText().equals("")){
59                     PengeluaranDana p = model.cariPengeluaran(keterangan);
60                     if(p!=null){
61                         view.getLabelKeterangan().setText(p.getKeterangan());
62                         view.getTfLihatIDPengeluaran().setText("");
63                     }else{
64                         JOptionPane.showMessageDialog(null, "ID not found, please fill the true ID");
65                     }
66                 }else{
67                     JOptionPane.showMessageDialog(null, "Please fill ID Pengeluaran");
68                 }
69             }else if (source == view.getBtnAddPengajuan()){
70                 if(!view.getTaTujuanPengajuan().getText().equals("") && !view.getTfTotalPengajuan().getText().equals("")){
71                     String idPengajuan = view.getTfPengajuan().getText();
72                     String idFakultas = fakultas.getIdFakultas();
73                     String tujuanPengajuan = view.getTaTujuanPengajuan().getText();
74                     int totalPengajuan = Integer.parseInt(view.getTfTotalPengajuan().getText());
75                     PengajuanDana pj = new PengajuanDana(idPengajuan, idFakultas, tujuanPengajuan, totalPengajuan, "0");
76                     model.addPengajuanDana(pj); 
77                     JOptionPane.showMessageDialog(null, "Add Data Success");
78                     //setText
79                     view.getTaTujuanPengajuan().setText("");
80                     view.getTfTotalPengajuan().setText("");
81                 }else{
82                     JOptionPane.showMessageDialog(null, "Plese fill all the blank");
83                 }
84             }else if (source == view.getBtnLogoutFakultas()){
85                 new ControllerLogin(model);
86                 view.dispose();
87             }
88             model.showTabelPengeluaranDanaFakultas(view, fakultas.getIdFakultas());
89             view.getTfDana().setText(Integer.toString(model.showDanaSisa(fakultas.getIdFakultas())));
90             view.refresh("");
91             view.SetIDPengajuan("");
92             model.showTablePengajuanDanaFakultas(view, fakultas.getIdFakultas());
93         }
