22     public class ControllerMenuFakultas implements ActionListener{
23         private Aplikasi model;
24         private MenuFakultas view;
25         private Database data = new Database();
26         private Fakultas fakultas;
27     
28         ControllerMenuFakultas(Aplikasi model, Fakultas f) {
29             this.model = model;
30             this.fakultas = f;
31             data.connect();
32             this.view = new MenuFakultas();
33             view.setVisible(true);
34             view.addListener(this);
35             view.refresh("");
36             view.getTfDana().setText(Integer.toString(model.showDanaSisa(fakultas.getIdFakultas())));
37             model.showTabelPembagianDanaFakultas(view,fakultas.getIdFakultas());
38             model.showTabelPengeluaranDanaFakultas(view, fakultas.getIdFakultas());
39             model.showTablePengajuanDanaFakultas(view, fakultas.getIdFakultas());
40             view.setIDPengajuan("");
41         }
42         
43         @Override
44         public void actionPerformed(ActionEvent ae) {
45         Object source = ae.getSource();
46            
47             if (source == view.getBtnAddPengeluaran()){          
48                 if(!"".equals(view.getTaKeterangan().getText())&&!"".equals(view.getTfTotalPengeluaran().getText())){
49                     String idPengeluran = view.getTfIDPengeluaran().getText();
50                     String tahunAjar = (String)view.getComboTahunAjar().getSelectedItem();
51                     String taKeterangan = view.getTaKeterangan().getText();
52                     int totalPengeluaran = Integer.parseInt(view.getTfTotalPengeluaran().getText());
53                     if (model.showDanaSisa(fakultas.getIdFakultas())-totalPengeluaran>0){
54                         PengeluaranDana p = new PengeluaranDana(idPengeluran, fakultas.getIdFakultas(), tahunAjar, taKeterangan, totalPengeluaran);
55                         model.tambahPengeluaranFakultas(p);
56                         JOptionPane.showMessageDialog(null, "Input Success");
57                     }else{
58                         JOptionPane.showMessageDialog(null, "Not enough Dana that we have");
59                     } 
60                     view.getTaKeterangan().setText("");
61                     view.getTfTotalPengeluaran().setText("");
62                 }else{
63                     JOptionPane.showMessageDialog(null, "Fill all the data");
64                 }           
65             }else if (source == view.getBtnLihatKeterangan()){
66                 String keterangan = view.getTfLihatIDPengeluaran().getText();
67                 if(!"".equals(view.getTfLihatIDPengeluaran().getText())){
68                     PengeluaranDana p = model.cariPengeluaran(keterangan);
69                     if(p!=null){
70                         view.getLabelKeterangan().setText(p.getKeterangan());
71                         view.getTfLihatIDPengeluaran().setText("");
72                     }else{
73                         JOptionPane.showMessageDialog(null, "ID not found, please fill the true ID");
74                     }
75                 }else{
76                     JOptionPane.showMessageDialog(null, "Please fill ID Pengeluaran");
77                 }
78             }else if (source == view.getBtnAddPengajuan()){
79                 if(!"".equals(view.getTaTujuanPengajuan().getText()) && !"".equals(view.getTfTotalPengajuan().getText())){
80                     String idPengajuan = view.getTfPengajuan().getText();
81                     String idFakultas = fakultas.getIdFakultas();
82                     String tujuanPengajuan = view.getTaTujuanPengajuan().getText();
83                     int totalPengajuan = Integer.parseInt(view.getTfTotalPengajuan().getText());
84                     PengajuanDana pj = new PengajuanDana(idPengajuan, idFakultas, tujuanPengajuan, totalPengajuan, "0");
85                     model.addPengajuanDana(pj); 
86                     JOptionPane.showMessageDialog(null, "Add Data Success");
87                     //setText
88                     view.getTaTujuanPengajuan().setText("");
89                     view.getTfTotalPengajuan().setText("");
90                 }else{
91                     JOptionPane.showMessageDialog(null, "Plese fill all the blank");
92                 }
93             }else if (source == view.getBtnLogoutFakultas()){
94                 new ControllerLogin(model);
95                 view.dispose();
96             }
97             model.showTabelPengeluaranDanaFakultas(view, fakultas.getIdFakultas());
98             view.getTfDana().setText(Integer.toString(model.showDanaSisa(fakultas.getIdFakultas())));
99             view.refresh("");
100             view.setIDPengajuan("");
101             model.showTablePengajuanDanaFakultas(view, fakultas.getIdFakultas());
102         }
