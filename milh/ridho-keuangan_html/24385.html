22     public class ControllerMenuWakilRektor implements ActionListener {
23         private Aplikasi model;
24         private MenuWakilRektor view;
25         private Database data = new Database();
26         
27         ControllerMenuWakilRektor(Aplikasi model){
28             this.model = model;
29             data.connect();
30             this.view = new MenuWakilRektor();
31             view.setVisible(true);
32             view.addListener(this);
33             view.refresh("");
34             model.showTabelPengajuan(view);
35             model.showTabelDana(view);
36             model.showTabelPembagianWR(view);
37             view.getTfDanaUniversitas().setText(Integer.toString(model.showDana()));
38         }
39         
40         
41         @Override
42         public void actionPerformed(ActionEvent ae) {
43             Object source = ae.getSource();
44             if (source == view.getBtnAddKonfirmasiPengajuan()){
45                 if ("".equals(view.getTfIDPengajuanWR().getText())){
46                     JOptionPane.showMessageDialog(null, "Fill the ID");
47                 } else {
48                     PengajuanDana pd = model.cariIDPengajuan(view.getTfIDPengajuanWR().getText());
49                     if (pd != null){
50                         if (view.getRbSetujuPengajuan().isSelected()){
51                             if (pd.getTotal() > Integer.parseInt(view.getTfDanaUniversitas().getText())){
52                                 JOptionPane.showMessageDialog(null, "Sorry insufficient funds!");
53                             } else {
54                                 model.konfirmasiPengajuan(view.getTfIDPengajuanWR().getText(), "1");
55                             }
56                         } else {
57                             model.konfirmasiPengajuan(view.getTfIDPengajuanWR().getText(), "-1");
58                         }
59                         
60                     } else {
61                         JOptionPane.showMessageDialog(null, "Sorry ID can't be found!");
62                     }
63                 }
64             } else if (source == view.getBtnAddPembagianDana()){
65                 if ("".equals(view.getTfFakultas().getText()) || "".equals(view.getTfTotaLDana().getText())){
66                     JOptionPane.showMessageDialog(null, "Fill all the textfield!");
67                 } else {
68                     if (model.cariFakultas(view.getTfFakultas().getText()) != null){
69                         PembagianDana pd = new PembagianDana(view.getTfPembagian().getText(), view.getTfFakultas().getText(), Integer.parseInt(view.getTfTotaLDana().getText()));
70                         Fakultas f = model.cariFakultas(view.getTfFakultas().getText());
71                         if (pd.getTotalDana() > Integer.parseInt(view.getTfDanaUniversitas().getText())){
72                             JOptionPane.showMessageDialog(null, "Sorry insufficient funds!");
73                         } else {
74                             model.tambahPembagianDana(pd);
75                             model.updateDanaFakultas(pd, f);
76                         }
77                     }
78                 }
79             } else if (source == view.getBtnLogout()){
80                 new ControllerLogin(model);
81                 view.dispose();
82             }
83             model.showTabelPengajuan(view);
84             view.getTfDanaUniversitas().setText(Integer.toString(model.showDana()));
85             view.refresh("");
86         }
87     }
